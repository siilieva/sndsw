383 Bool_t Scifi::ProcessHits(FairVolume* vol)
384 {
386  //Set parameters at entrance of volume. Reset ELoss.
387  if ( gMC->IsTrackEntering() )
388  {
389  fELoss = 0.;
390  fTime = gMC->TrackTime() * 1.0e09;
391  fLength = gMC->TrackLength();
392  gMC->TrackPosition(fPos);
393  gMC->TrackMomentum(fMom);
394  TGeoNavigator* nav = gGeoManager->GetCurrentNavigator();
395 
396  int fibre_local_id = nav->GetMother()->GetNumber() - 1e6 - 1e4; // Local ID within the mat.
397  int fibre_mat_id = nav->GetMother(2)->GetNumber(); // Get mat ID.
398 
399  int fibre_station_number = int( fibre_mat_id / 1e6); // Get the station number from the mat
400 
401  int fibre_mat_id_station_removed = fibre_mat_id - fibre_station_number*1e6;
402  int fibre_mat_number = int((fibre_mat_id_station_removed - int(fibre_mat_id_station_removed/1e5)*1e5)/1e4);
403 
404  fVolumeID = fibre_local_id + fibre_station_number*1e6 + fibre_mat_number*1e4;
405 
406  if (fVolumeID==0){std::cout<<"fiber vol id "<<nav->GetMother()->GetName()<<std::endl;}
407 
408  }
409  // Sum energy loss for all steps in the active volume
410  fELoss += gMC->Edep();
411 
412  // Create ScifiPoint at exit of active volume
413  if ( gMC->IsTrackExiting() ||
414  gMC->IsTrackStop() ||
415  gMC->IsTrackDisappeared() ) {
416  if (fELoss == 0. ) { return kFALSE; }
417  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
418 /* STMRFFF
419 First digit S: station # within the sub-detector
420 Second digit T: type of the plane: 0-horizontal fiber plane, 1-vertical fiber plane
421 Third digit M: determines the mat number
422 Fourth digit R: row number (in Z direction)
423 Last three digits F: fiber number
424 */
425  TParticle* p=gMC->GetStack()->GetCurrentTrack();
426  Int_t pdgCode = p->GetPdgCode();
427 
428  TLorentzVector Pos;
429  gMC->TrackPosition(Pos);
430  Double_t xmean = (fPos.X()+Pos.X())/2. ;
431  Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
432  Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
433 
434  AddHit(fTrackID,fVolumeID, TVector3(xmean, ymean, zmean),
435  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
436  fELoss, pdgCode);
437 
438  // Increment number of det points in TParticle
439  ShipStack* stack = (ShipStack*) gMC->GetStack();
440  stack->AddPoint(kLHCScifi);
441  }
442 
443  return kTRUE;

599 Double_t Scifi::area(Double_t a,Double_t R,Double_t xL,Double_t xR)
600 {
601  Double_t fracL = -1;
602  Double_t fracR = -1;
603  if (xL<=a-R && xR>=a+R) {return 1;}
604  Double_t leftC = ycross(a,R,xL);
605  Double_t rightC = ycross(a,R,xR);
606  if (leftC<0 && rightC<0) {return -1;}
607  if ( !(rightC<0) ){ fracR = fraction(R,abs(xR-a),rightC);}
608  if ( !(leftC<0) ) { fracL = fraction(R,abs(xL-a),leftC);}
609  Double_t theAnswer = 0;
610  if ( !(leftC<0) ) {
611  if(xL<a){theAnswer += 1-fracL;}
612  else{ theAnswer += fracL;}
613  if ( !(rightC<0) ) {theAnswer -=1;}
614  }
615  if ( !(rightC<0) ){
616  if(xR>a){ theAnswer += 1-fracR;}
617  else{ theAnswer += fracR;}
618  }
619  return theAnswer;
620 }

622 void Scifi::SiPMmapping(){
623  Float_t fibresRadius = -1;
624  Float_t dSiPM = -1;
625  TGeoNode* vol;
626  TGeoNode* fibre;
627  SiPMOverlap(); // 12 SiPMs per mat, made for horizontal mats, fibres staggered along y-axis.
628  auto sipm = gGeoManager->FindVolumeFast("SiPMmapVol");

632  auto mat = static_cast<TGeoNode*>(plane->GetNodes()->At(imat));
633  Float_t t1 = mat->GetMatrix()->GetTranslation()[1];
634  auto vmat = mat->GetVolume();
635  for (int ifibre = 0; ifibre < vmat->GetNodes()->GetEntriesFast(); ifibre++){
636  fibre = static_cast<TGeoNode*>(vmat->GetNodes()->At(ifibre));
637  if (fibresRadius<0){
638  auto tmp = fibre->GetVolume()->GetShape();
639  auto S = dynamic_cast<TGeoBBox*>(tmp);
640  fibresRadius = S->GetDX();
641  }
642  Float_t t2 = fibre->GetMatrix()->GetTranslation()[1];
643  Int_t fID = fibre->GetNumber()%100000 + imat*1e4; // local fibre number, global fibre number = SO+fID
644  Float_t a = t1+t2;
645 
646  // check for overlap with any of the SiPM channels in the same mat
647  for(Int_t nChan = 0; nChan< Nodes->GetEntriesFast();nChan++){ // 12 SiPMs total and 4 SiPMs per mat times 128 channels
648  vol = static_cast<TGeoNode*>(Nodes->At(nChan));
649  Int_t N = vol->GetNumber()%100000;
650  if (imat!=int(N/10000)){continue;}
651  Float_t xcentre = vol->GetMatrix()->GetTranslation()[1];
652  if (dSiPM<0){
653  TGeoBBox* B = dynamic_cast<TGeoBBox*>(vol->GetVolume()->GetShape());
654  dSiPM = B->GetDY();
655  }
656  if (TMath::Abs(xcentre-a)>4*fibresRadius){ continue;} // no need to check further
657  Float_t W = area(a,fibresRadius,xcentre-dSiPM,xcentre+dSiPM);
658  if (W<0){ continue;}

660  Wa[0] = W;

663  }
664  }
665  }

667  // probably an overkill, maximum difference between weighted average and central position < 6 micron.
668  std::map<Int_t,std::map<Int_t,std::array<float, 2>>>::iterator it;
669  std::map<Int_t,std::array<float, 2>>::iterator itx;
670  for (it = fibresSiPM.begin(); it != fibresSiPM.end(); it++)
