7 def hit_finder(slope, intercept, box_centers, box_ds, tol = 0.) :
8  """ Finds hits intersected by Hough line """
9 
10  # First check if track at center of box is within box limits
11  d = np.abs(box_centers[0,:,1] - (box_centers[0,:,0]*slope + intercept))
12  center_in_box = d < (box_ds[0,:,1]+tol)/2.
13 
14  # Now check if, assuming line is not in box at box center, the slope is large enough for line to clip the box at corner
15  clips_corner = np.abs(slope) > np.abs((d - (box_ds[0,:,1]+tol)/2.)/(box_ds[0,:,0]+tol)/2.)
16 
17  # If either of these is true, line goes through hit:
18  hit_mask = np.logical_or(center_in_box, clips_corner)
19 
20  # Return indices
21  return np.where(hit_mask)[0]

48  def fit(self, hit_collection, draw = False, weights = None) :
49 
50  self.accumulator = np.zeros((self.n_r, self.n_theta))
51  for i_hit, hit in enumerate(hit_collection) :
52  hit_r = hit[0]*self.cos_thetas + hit[1]*self.sin_thetas
53  out_of_range = np.logical_and(hit_r > self.r_range[0], hit_r < self.r_range[1])
54  hit_r_i = np.floor((hit_r[out_of_range] - self.r_range[0])/(self.r_range[1] - self.r_range[0])*self.n_r).astype(np.int)
55 
56  if weights is not None :
57  self.accumulator[hit_r_i, self.theta_i[out_of_range]] += weights[i_hit]
58  else :
59  self.accumulator[hit_r_i, self.theta_i[out_of_range]] += 1
60 
61  # Smooth accumulator
62  if self.smooth :
63  self.accumulator = scipy.ndimage.gaussian_filter(self.accumulator, 3)
64 
65  # This might be useful for debugging, but leave out for now.
66  if draw :
67  pass

75 
76  found_r = self.r_bins[i_max[0]]
77  found_theta = self.theta_bins[i_max[1]]
