220def sum_min_dca(event) :
221 if len(event.Reco_MuonTracks) < 1 :
222 return False, 999
223
224 track = event.Reco_MuonTracks.At(0)

226 track_start = track.getStart()
227 track_stop = track.getStop()
228
229 slope_X = (track_stop.X() - track_start.X())/(track_stop.Z() - track_start.Z())

242 scifiDet.GetSiPMPosition(hit.GetDetectorID(), a, b)

244 if hit.isVertical() :
245 slope_XY = slope_X
246 track_start_XY = track_start.X()
247 track_stop_XY = track_stop.X()
248 hit_pos = [(a.Z() + b.Z())/2., (a.X() + b.X())/2.]
249 else :
250 slope_XY = slope_Y
251 track_start_XY = track_start.Y()

255 if hit.isVertical() :

258 min_dca_ver[hit.GetStation()-1] = this_d

262 min_dca_hor[hit.GetStation()-1] = this_d

264 min_dca_ver = np.array(min_dca_ver)
