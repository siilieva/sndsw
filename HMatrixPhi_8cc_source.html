89void HMatrixPhi::HMHt(TMatrixDSym& M) const {
90 assert (M.GetNrows() == 5);
91
92 M(0,0) = cosPhi_ * (cosPhi_*M(3,3) + sinPhi_*M(3,4))
93 + sinPhi_ * (cosPhi_*M(4,3) + sinPhi_*M(4,4));
94
95 M.ResizeTo(1,1);
96}
