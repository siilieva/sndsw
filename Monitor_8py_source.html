589  def langaufun(self,x,par):
590  #Fit parameters:
591  #par[0]=Width (scale) parameter of Landau density
592  #par[1]=Most Probable (MP, location) parameter of Landau density
593  #par[2]=Total area (integral -inf to inf, normalization constant)
594  #par[3]=Width (sigma) of convoluted Gaussian function
595  #
596  #In the Landau distribution (represented by the CERNLIB approximation),
597  #the maximum is located at x=-0.22278298 with the location parameter=0.
598  #This shift is corrected within this function, so that the actual
599  #maximum is identical to the MP parameter.
600 #
601  # Numeric constants
602  invsq2pi = 0.3989422804014 # (2 pi)^(-1/2)
603  mpshift = -0.22278298 # Landau maximum location
604 #
605  # Control constants
606  np = 100.0 # number of convolution steps
607  sc = 5.0 # convolution extends to +-sc Gaussian sigmas
608 #
609  # Variables
610  summe = 0.0
611 #
612  # MP shift correction
613  mpc = par[1] - mpshift * par[0]
614 #
615  # Range of convolution integral
616  xlow = max(0,x[0] - sc * par[3])
617  xupp = x[0] + sc * par[3]
618 #
619  step = (xupp-xlow) / np
620 #
621  # Convolution integral of Landau and Gaussian by sum
622  i=1.0
623  if par[0]==0 or par[3]==0: return 9999
624  while i<=np/2:
625  i+=1
626  xx = xlow + (i-.5) * step
627  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
628  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
629 #
630  xx = xupp - (i-.5) * step
631  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
632  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
633 #
634  return (par[2] * step * summe * invsq2pi / par[3])
