496  def langaufun(self,x,par):
497  #Fit parameters:
498  #par[0]=Width (scale) parameter of Landau density
499  #par[1]=Most Probable (MP, location) parameter of Landau density
500  #par[2]=Total area (integral -inf to inf, normalization constant)
501  #par[3]=Width (sigma) of convoluted Gaussian function
502  #
503  #In the Landau distribution (represented by the CERNLIB approximation),
504  #the maximum is located at x=-0.22278298 with the location parameter=0.
505  #This shift is corrected within this function, so that the actual
506  #maximum is identical to the MP parameter.
507 #
508  # Numeric constants
509  invsq2pi = 0.3989422804014 # (2 pi)^(-1/2)
510  mpshift = -0.22278298 # Landau maximum location
511 #
512  # Control constants
513  np = 100.0 # number of convolution steps
514  sc = 5.0 # convolution extends to +-sc Gaussian sigmas
515 #
516  # Variables
517  summe = 0.0
518 #
519  # MP shift correction
520  mpc = par[1] - mpshift * par[0]
521 #
522  # Range of convolution integral
523  xlow = max(0,x[0] - sc * par[3])
524  xupp = x[0] + sc * par[3]
525 #
526  step = (xupp-xlow) / np
527 #
528  # Convolution integral of Landau and Gaussian by sum
529  i=1.0
530  if par[0]==0 or par[3]==0: return 9999
531  while i<=np/2:
532  i+=1
533  xx = xlow + (i-.5) * step
534  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
535  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
536 #
537  xx = xupp - (i-.5) * step
538  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
539  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
540 #
541  return (par[2] * step * summe * invsq2pi / par[3])
