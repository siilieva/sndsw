602  def langaufun(self,x,par):
603  #Fit parameters:
604  #par[0]=Width (scale) parameter of Landau density
605  #par[1]=Most Probable (MP, location) parameter of Landau density
606  #par[2]=Total area (integral -inf to inf, normalization constant)
607  #par[3]=Width (sigma) of convoluted Gaussian function
608  #
609  #In the Landau distribution (represented by the CERNLIB approximation),
610  #the maximum is located at x=-0.22278298 with the location parameter=0.
611  #This shift is corrected within this function, so that the actual
612  #maximum is identical to the MP parameter.
613 #
614  # Numeric constants
615  invsq2pi = 0.3989422804014 # (2 pi)^(-1/2)
616  mpshift = -0.22278298 # Landau maximum location
617 #
618  # Control constants
619  np = 100.0 # number of convolution steps
620  sc = 5.0 # convolution extends to +-sc Gaussian sigmas
621 #
622  # Variables
623  summe = 0.0
624 #
625  # MP shift correction
626  mpc = par[1] - mpshift * par[0]
627 #
628  # Range of convolution integral
629  xlow = max(0,x[0] - sc * par[3])
630  xupp = x[0] + sc * par[3]
631 #
632  step = (xupp-xlow) / np
633 #
634  # Convolution integral of Landau and Gaussian by sum
635  i=1.0
636  if par[0]==0 or par[3]==0: return 9999
637  while i<=np/2:
638  i+=1
639  xx = xlow + (i-.5) * step
640  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
641  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
642 #
643  xx = xupp - (i-.5) * step
644  fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
645  summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
646 #
647  return (par[2] * step * summe * invsq2pi / par[3])
