520 bar = 4

528 self.MuFilter.GetPosition(s*10000+p*1000+bar,A,B)

531 mat = 1

536 zPos['Scifi'][s*10+o] = (A.Z()+B.Z())/2.

610 def langaufun(self,x,par):
611 #Fit parameters:
612 #par[0]=Width (scale) parameter of Landau density
613 #par[1]=Most Probable (MP, location) parameter of Landau density
614 #par[2]=Total area (integral -inf to inf, normalization constant)
615 #par[3]=Width (sigma) of convoluted Gaussian function
616 #
617 #In the Landau distribution (represented by the CERNLIB approximation),
618 #the maximum is located at x=-0.22278298 with the location parameter=0.
619 #This shift is corrected within this function, so that the actual
620 #maximum is identical to the MP parameter.
621#
622 # Numeric constants
623 invsq2pi = 0.3989422804014 # (2 pi)^(-1/2)
624 mpshift = -0.22278298 # Landau maximum location
625#
626 # Control constants
627 np = 100.0 # number of convolution steps
628 sc = 5.0 # convolution extends to +-sc Gaussian sigmas
629#
630 # Variables
631 summe = 0.0
632#
633 # MP shift correction
634 mpc = par[1] - mpshift * par[0]
635#
636 # Range of convolution integral
637 xlow = max(0,x[0] - sc * par[3])
638 xupp = x[0] + sc * par[3]
639#
640 step = (xupp-xlow) / np
641#
642 # Convolution integral of Landau and Gaussian by sum
643 i=1.0
644 if par[0]==0 or par[3]==0: return 9999
645 while i<=np/2:
646 i+=1
647 xx = xlow + (i-.5) * step
648 fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
649 summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
650#
651 xx = xupp - (i-.5) * step
652 fland = ROOT.TMath.Landau(xx,mpc,par[0]) / par[0]
653 summe += fland * ROOT.TMath.Gaus(x[0],xx,par[3])
654#
655 return (par[2] * step * summe * invsq2pi / par[3])
