48 InitStatus DigiTaskSND::Init()
49 {
50  FairRootManager* ioman = FairRootManager::Instance();
51  if (!ioman) {
52  cout << "-E- DigiTaskSND::Init: "
53  << "RootManager not instantiated!" << endl;
54  return kFATAL;
55  }
56 
57  // Get the SciFi detector and sipm to fibre mapping
58  scifi = dynamic_cast<Scifi*> (gROOT->GetListOfGlobals()->FindObject("Scifi") );
59  scifi->SiPMmapping();
60  fibresSiPM = scifi->GetSiPMmap();
61  siPMFibres = scifi->GetFibresMap();
62 
63  // Get event header
64  // Try classic FairRoot approach first
65  fMCEventHeader = static_cast<FairMCEventHeader*> (ioman->GetObject("MCEventHeader."));
66  // .. with a safety net for trailing dots mischief
67  if ( fMCEventHeader == nullptr ) {
68  fMCEventHeader = static_cast<FairMCEventHeader*>(gROOT->FindObjectAny("MCEventHeader."));
69  }
70  // Get input MC points
71  fScifiPointArray = static_cast<TClonesArray*>(ioman->GetObject("ScifiPoint"));
72  fvetoPointArray = static_cast<TClonesArray*>(ioman->GetObject("vetoPoint"));
73  fEmulsionPointArray = static_cast<TClonesArray*>(ioman->GetObject("EmulsionDetPoint"));
74  fMuFilterPointArray = static_cast<TClonesArray*>(ioman->GetObject("MuFilterPoint"));
75  if (!fScifiPointArray and !fMuFilterPointArray) {
76  cout << "-W- DigiTaskSND::Init: "
77  << "No Scifi and no MuFilter MC Point array!" << endl;
78  return kERROR;
79  }
80  // copy branches from input file:
81  fMCTrackArray = static_cast<TClonesArray*>(ioman->GetObject("MCTrack"));
82  ioman->Register("MCTrack", "ShipMCTrack", fMCTrackArray, kTRUE);
83  ioman->Register("vetoPoint", "vetoPoints", fvetoPointArray, kTRUE);
84  ioman->Register("EmulsionDetPoint", "EmulsionDetPoints", fvetoPointArray, kTRUE);
85  ioman->Register("ScifiPoint", "ScifiPoints", fScifiPointArray, kTRUE);
86  ioman->Register("MuFilterPoint", "MuFilterPoints", fMuFilterPointArray, kTRUE);
87  ioman->RegisterAny("MCEventHeader.", fMCEventHeader, kTRUE);
88 
89  // Event header
90  fEventHeader = new FairEventHeader();
91  ioman->Register("EventHeader", "sndEventHeader", fEventHeader, kTRUE);
92 
93  // Create and register output array - for SciFi and MuFilter
94  fScifiDigiHitArray = new TClonesArray("sndScifiHit");
95  ioman->Register("Digi_ScifiHits", "DigiScifiHit_det", fScifiDigiHitArray, kTRUE);
96  // Branche containing links to MC truth info
97  fScifiHit2MCPointsArray = new TClonesArray("Hit2MCPoints");
98  ioman->Register("Digi_ScifiHits2MCPoints", "DigiScifiHits2MCPoints_det", fScifiHit2MCPointsArray, kTRUE);
99  fScifiHit2MCPointsArray->BypassStreamer(kTRUE); fScifiClusterArray = new TClonesArray("sndCluster");
100  ioman->Register("Cluster_Scifi", "ScifiCluster_det", fScifiClusterArray, kTRUE);
101  fMuFilterDigiHitArray = new TClonesArray("MuFilterHit");
102  ioman->Register("Digi_MuFilterHits", "DigiMuFilterHit_det", fMuFilterDigiHitArray, kTRUE);
103  // Branche containing links to MC truth info
104  fMuFilterHit2MCPointsArray = new TClonesArray("Hit2MCPoints");
105  ioman->Register("Digi_MuFilterHits2MCPoints", "DigiMuFilterHits2MCPoints_det", fMuFilterHit2MCPointsArray, kTRUE);
106  fMuFilterHit2MCPointsArray->BypassStreamer(kTRUE);
107 
108  ScifiThreshold = 3.5;
109  MufiLargeThreshold = 0;
110  MufiSmallThreshold = 0;
111 
112  return kSUCCESS;
113 }

137 void DigiTaskSND::digitizeScifi()
138 {
139  // a map containing fibreID and vector(list) of points and weights
140  map<int, pair<vector<ScifiPoint*>, vector<float>> > hitContainer{};
141  Hit2MCPoints mcLinks;
142  map<pair<int, int>, double> mcPoints{};
143  map<int, double> norm{};
144  int globsipmChan{}, detID{};
145  int locFibreID{};
146 
147  // Fill the map
148  for (int k = 0, kEnd = fScifiPointArray->GetEntries(); k < kEnd; k++)
149  {
150  ScifiPoint* point = static_cast<ScifiPoint*>(fScifiPointArray->At(k));
151  if (!point) continue;
152  // Collect all hits in same SiPM channel
153  detID = point->GetDetectorID();
154  locFibreID = detID%100000;
155  // Check if locFibreID in a dead area
156  if(siPMFibres.count(locFibreID)==0) continue;
157  double dE{};
158  float weight{};
159  for (auto sipmChan : siPMFibres[locFibreID])
160  {
161  globsipmChan = int(detID/100000)*100000+sipmChan.first;
162  // Initializing - not needed in C++
163  /*if (hitContainer[globsipmChan].first.size()==0){
164  hitContainer[globsipmChan] = {};
165  mcPoints[make_pair(globsipmChan, k)] = {};
166  norm[globsipmChan] = {};
167  }*/
168  weight = sipmChan.second[0];
169  hitContainer[globsipmChan].first.push_back(point);
170  hitContainer[globsipmChan].second.push_back(weight);
171  dE = point->GetEnergyLoss()*weight;
172  mcPoints[make_pair(globsipmChan, k)] = dE;
173  norm[globsipmChan]+= dE;
174  }
175  }// End filling map
176  int index = 0;
177  // Loop over entries of the hitContainer map and collect all hits in same detector element
178  for (auto it = hitContainer.begin(); it != hitContainer.end(); it++){
179  new ((*fScifiDigiHitArray)[index]) sndScifiHit(it->first, hitContainer[it->first].first, hitContainer[it->first].second);
180  index++;
181  for (auto mcit = mcPoints.begin(); mcit != mcPoints.end(); mcit++){
182  if(it->first == mcit->first.first) mcLinks.Add(it->first, mcit->first.second, mcPoints[make_pair(it->first, mcit->first.second)]/norm[it->first]);
183  }
184  }
185  new((*fScifiHit2MCPointsArray)[0]) Hit2MCPoints(mcLinks);
186 }

188 void DigiTaskSND::clusterScifi()
189 {
190  map<int, int > hitDict{};
191  vector<int> hitList{};
192  vector<int> tmp{};
193  int index{}, ncl{}, cprev{}, c{}, last{}, first{}, N{};
194 
195  for (int k = 0, kEnd = fScifiDigiHitArray->GetEntries(); k < kEnd; k++) {
196  sndScifiHit* d = static_cast<sndScifiHit*>(fScifiDigiHitArray->At(k));
197  if (!d->isValid()) continue;
198  hitDict[d->GetDetectorID()] = k ;
199  hitList.push_back(d->GetDetectorID());
200  }
201  if (hitList.size() > 0)
202  {
203  sort(hitList.begin(), hitList.end());
204  tmp.push_back(hitList[0]);
205  cprev = hitList[0];
206  ncl = 0;
207  last = hitList.size()-1;
208  vector<sndScifiHit*> hitlist{};
209  for (int i =0; i<hitList.size(); i++)
210  {
211  if (i==0 && hitList.size()>1) continue;
212  c = hitList[i];
213  if (c-cprev ==1) tmp.push_back(c);
214  if (c-cprev !=1 || c==hitList[last]){
215  first = tmp[0];
216  N = tmp.size();
217  hitlist.clear();
218  for (int j=0; j<tmp.size(); j++)
219  {
220  sndScifiHit* aHit = static_cast<sndScifiHit*>(fScifiDigiHitArray->At(hitDict[tmp[j]]));
221  hitlist.push_back(aHit);
222  }
223  new ((*fScifiClusterArray)[index]) sndCluster(first, N, hitlist, scifi);
224  index++;
225  if (c!=hitList[last])
226  {
227  ncl++;
228  tmp.clear();
229  tmp.push_back(c);
230  }
231  }
232  cprev = c;
233  }
234  }
235 }
