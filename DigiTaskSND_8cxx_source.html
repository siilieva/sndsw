133  ScifiPoint* point = static_cast<ScifiPoint*>(fScifiPointArray->At(k));
134  if (!point) continue;
135  // Collect all hits in same SiPM channel
136  detID = point->GetDetectorID();

138  // Check if locFibreID in a dead area

140  double dE{};
141  float weight{};

144  globsipmChan = int(detID/100000)*100000+sipmChan.first;
145  weight = sipmChan.second[0];
146  hitContainer[globsipmChan].first.push_back(point);
147  hitContainer[globsipmChan].second.push_back(weight);
148  dE = point->GetEnergyLoss()*weight;
149  mcPoints[make_pair(globsipmChan, k)] = dE;
150  norm[globsipmChan]+= dE;
151  }
152  }// End filling map

155  for (auto it = hitContainer.begin(); it != hitContainer.end(); it++){

157  index++;

159  if(it->first == mcit->first.first) mcLinks.Add(it->first, mcit->first.second, mcPoints[make_pair(it->first, mcit->first.second)]/norm[it->first]);
