32 Double_t phi = gRandom->Uniform(0,2*TMath::Pi());

42 Double_t x = R*TMath::Sin(theta)*TMath::Cos(phi);
43 Double_t y = R*TMath::Sin(theta)*TMath::Sin(phi);
