312  def comb_calibration(self,board_id,tofpet_id,channel,tac,v_coarse,v_fine,t_coarse,t_fine,GQDC = 1.0, TDC=0): # max gain QDC = 3.6
313  qdc_cal = self.qdc_calqdc_cal
314  par = qdc_cal[board_id][tofpet_id][channel][tac]
315  parT = par[TDC]
316  x = t_fine
317  ftdc = (-parT['b']-ROOT.TMath.Sqrt(parT['b']**2-4*parT['a']*(parT['c']-x)))/(2*parT['a']) # Ettore 28/01/2022 +parT['d']
318  timestamp = t_coarse + ftdc
319  tf = timestamp - t_coarse
320  x = v_coarse - tf
321  fqdc = -par['c']*ROOT.TMath.Log(1+ROOT.TMath.Exp( par['a']*(x-par['e'])**2-par['b']*(x-par['e']) )) + par['d']
322  value = (v_fine-fqdc)/GQDC
323  return timestamp,value,max(par['chi2Ndof'],parT['chi2Ndof']),v_fine/par['d']
