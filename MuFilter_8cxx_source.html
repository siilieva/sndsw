403 Bool_t MuFilter::ProcessHits(FairVolume* vol)
404 {
406  //Set parameters at entrance of volume. Reset ELoss.
407  if ( gMC->IsTrackEntering() )
408  {
409  fELoss = 0.;
410  fTime = gMC->TrackTime() * 1.0e09;
411  fLength = gMC->TrackLength();
412  gMC->TrackPosition(fPos);
413  gMC->TrackMomentum(fMom);
414  }
415  // Sum energy loss for all steps in the active volume
416  fELoss += gMC->Edep();
417 
418  // Create MuFilterPoint at exit of active volume
419  if ( gMC->IsTrackExiting() ||
420  gMC->IsTrackStop() ||
421  gMC->IsTrackDisappeared() ) {

506 Float_t MuFilter::GetCorrectedTime(Int_t fDetectorID, Int_t channel, Double_t rawTime, Double_t L){
507 /* expect time in u.ns and path length to sipm u.cm */
508 /* calibration implemented only for DS! */
509 /* channel 0 left or top, channel 1 right */
510  if (fDetectorID<30000){
511  LOG(ERROR) << "MuFilter::GetCorrectedTime: not yet implemented for Veto and DS, no correction applied" ;
512  return rawTime;
513  }
514  TString tag = "";
515  vector<int> coveredRuns{};
516  if (eventHeader){
517  Int_t fRunNumber = eventHeader->GetRunId();
518  if (fRunNumber<1){
519  LOG(ERROR) << "MuFilter::GetCorrectedTime: non valid run number "<<fRunNumber;
520  return rawTime;
521  }
522  // Get available tags from the geometry file
523  std::string tag_string;
524  for (auto key : conf_floats){
525  tag_string = key.first.Data();
526  if (tag_string.find("MuFilter/DSTcorslopet_") != string::npos){
527  coveredRuns.push_back(stoi(tag_string.substr(tag_string.find("t_")+2)));
528  }
529  }
530  if (coveredRuns.size()!=0){
531  tag = "t_"+to_string(coveredRuns[coveredRuns.size()-1]);
532  for (int i=1; i<coveredRuns.size(); i++){
533  if (fRunNumber>=coveredRuns[i-1] && fRunNumber<coveredRuns[i]){
534  tag = "t_"+to_string(coveredRuns[i-1]);
535  }
536  }
537  //special case
538  if (fRunNumber<5193 && fRunNumber>5174) tag = "t_"+to_string(coveredRuns[0]);
539  }
540  else{
541  // allow reading older geo files with letter tags i.e. A, B, C
542  tag = "A";
543  if (fRunNumber>5116 && !(fRunNumber<5193 && fRunNumber>5174) ) {tag = "B";}
544  }
545  }
546  Float_t cor = rawTime;
547  int l = (fDetectorID-30000)/1000;
548  int ichannel60 = fDetectorID%1000;
549  int p;
550  if (channel==0 && ichannel60<30 ){p=0+l*6;}
551  if (channel==0 && ichannel60>29 && ichannel60<60 ){p=1+l*6;}
552  if (channel==1 && ichannel60<30 ){p=2+l*6;}
553  if (channel==1 && ichannel60>29 && ichannel60<60 ){p=3+l*6;}
554  if (channel==0 && ichannel60>59 && ichannel60<90 ){p=4+l*6;}
555  if (channel==0 && ichannel60>89 && ichannel60<120 ){p=5+l*6;}
556  if (l==3){p-=4;}
557  if (ichannel60>59) {ichannel60-=60;}
558  // DS time alignment first order
559  if (ichannel60<30){cor += conf_floats["MuFilter/DSTcorslope"+tag]*(ichannel60-15);}
560  else{ cor -= conf_floats["MuFilter/DSTcorslope"+tag]*(ichannel60-45);}
561  string si = to_string(p);
562  cor -= conf_floats["MuFilter/DSTcorC"+si+tag];
563  cor -= L/conf_floats["MuFilter/DsPropSpeed"];
564  return cor;
565 }
