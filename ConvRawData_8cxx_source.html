889double ConvRawData::qdc_calibration( int board_id, int tofpet_id, int channel,
890 int tac, uint16_t v_coarse, uint16_t v_fine, uint16_t tf )
891 {
892 double GQDC = 1.0; // or 3.6
893 map<string, double> par= X_qdc[{board_id,tofpet_id,channel,tac}];
894 double x = v_coarse - tf;
895 double fqdc = -par["c"]*log(1+exp( par["a"]*pow((x-par["e"]),2)

897 double value = (v_fine-fqdc)/GQDC;
898 return value;

911double ConvRawData::time_calibration( int board_id, int tofpet_id, int channel,
912 int tac, int64_t t_coarse, uint16_t t_fine, int TDC=0 )
913{
914 map<string, double> parT = X_tdc[{board_id,tofpet_id,channel,tac, TDC}];
915 double x = t_fine;
916 double ftdc = (-parT["b"]-sqrt(pow(parT["b"],2)
917 -4*parT["a"]*(parT["c"]-x)))/(2*parT["a"]);
918 double timestamp = t_coarse+ftdc;
919 return timestamp;

922{
923 map<string, double> par= X_qdc[{board_id,tofpet_id,channel,tac}];
924 map<string, double> parT = X_tdc[{board_id,tofpet_id,channel,tac,TDC}];

926 double ftdc = (-parT["b"]-sqrt(pow(parT["b"],2)
