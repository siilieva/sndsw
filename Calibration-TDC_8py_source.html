59def extractMeanAndSigma(tdcCanvas,s):
60 C = {}
61 f = ROOT.TFile.Open(tdcCanvas)
62 tTDCcalib = f.Get('tTDCcalib_'+sdict[s]).Clone()
63 for pad in tTDCcalib.GetListOfPrimitives():
