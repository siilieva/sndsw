30 auto mf = reinterpret_cast<DataFrame *>(data);
31 total_size -= sizeof(DataFrame);
32 data = reinterpret_cast<Int_t *>(&(mf->hits));
33 auto frameTime = mf->header.frameTime;
34 while (total_size > 0) {
