86def docaLine(a,b,c,d):
87 u = b-a
88 v = d-c
89 n = u.Cross(v)
90 ac = a-c
91 if n.Mag()==0:
92 dist = ROOT.TMath.Sqrt(ac.Mag2()+3*(ac.Dot(u))**2/u.Mag2())
93 else:
94 dist = n.Dot(ac)/n.Mag()
95 return dist
